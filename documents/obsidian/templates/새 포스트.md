<%*
const title = await tp.system.prompt("글 제목");
const slug = await tp.system.prompt("슬러그 (파일명용 영문, 예: godot-signal)");
const category = await tp.system.suggester(
  ["study (공부)", "log (로그)", "life (일상)", "project (프로젝트)", "notice (공지)"],
  ["study", "log", "life", "project", "notice"],
  false,
  "카테고리 선택"
);
const series = await tp.system.prompt("시리즈 (폴더명, 없으면 비워두기)", "");
const date = tp.date.now("YYYY-MM-DD");
const fileName = `${date}-${slug}`;
const targetDir = series ? `content/posts/${category}/${series}` : `content/posts/${category}`;
await tp.file.move(`${targetDir}/${fileName}`);
-%>
